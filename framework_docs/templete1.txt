<script>
var f = document.frm;

function check_form()
{
	if (isBlank(f.username.value))
	{
		alert("이름을 입력해주세요.");
		f.username.focus();
		return false;
	}

	if (isBlank(f.hp.value))
	{
		alert("연락처를 입력해주세요.");
		f.hp.focus();
		return false;
	}
	if (isBlank(f.email.value))
	{
		alert("이메일을 입력해주세요.");
		f.email.focus();
		return false;
	}
	if(chkType(f.email.value, 8) == "resultFalse")
	{
		alert("이메일 형식을 확인해 주세요.");
		f.email.focus();
		return false;
	}
	if (isBlank(f.join_type.value))
	{
		alert("가맹형태를 선택해주세요.");
		f.join_type.focus();
		return false;
	}
	if (isBlank(f.hope_area.value))
	{
		alert("창업희망지역을 입력해주세요.");
		f.hope_area.focus();
		return false;
	}
	if (isBlank(f.startups_date.value))
	{
		alert("창업예정시기를 입력해주세요.");
		f.startups_date.focus();
		return false;
	}
	if (isBlank(f.estimated_cost.value))
	{
		alert("예상창업비용을 입력해주세요.");
		f.estimated_cost.focus();
		return false;
	}
	if ($("input[name='ownership_flag']").is(':checked') == false)
	{
		alert("현재점포소유를 선택해주세요.");
		$('input:radio[name="ownership_flag"]:eq(0)').focus();
		return false;
	}
	if (isBlank(f.comment.value))
	{
		alert("내용을 입력해주세요.");
		f.comment.focus();
		return false;
	}
	if (isBlank(f.antiSpamPw.value))
	{
		alert("자동등록방지코드를 입력해주세요.");
		f.antiSpamPw.focus();
		return false;
	}
	if ($("#agree1").is(":checked") == false)
	{
		alert("개인정보취급방침에 동의해주세요..");
		return false;
	}
	else
	{
		f.submit();
	}
}
</script>
